#!/usr/bin/env python3 """ Photo Sorter - Folder-based GBP photo tracker Tracks which photos have been used vs unused across brands, cities, neighborhoods. Usage: python photo_sorter.py setup # First-time setup python photo_sorter.py import # Import photos from _inbox/ python photo_sorter.py list # List unused photos python photo_sorter.py used # Mark a photo as used python photo_sorter.py stats # Show inventory + last upload dates python photo_sorter.py search # Search by tag python photo_sorter.py reassign # Fix brand/city/neighborhood python photo_sorter.py dupes # Scan library for duplicate photos python photo_sorter.py last-uploads # Show most recent upload per location python photo_sorter.py menu # Interactive menu (recommended) """ import os import sys import shutil import sqlite3 import re import hashlib from datetime import datetime from pathlib import Path # ============================================================ # CONFIGURATION - Edit this path to point to your photo library # ============================================================ LIBRARY_ROOT = Path.home() / "PhotoLibrary" # Your brands and their valid markets/neighborhoods BRANDS = { "mosquito-squad": { "chattanooga-tn": ["northshore", "hixson", "ooltewah", "east-brainerd", "signal-mountain"], "knoxville-tn": ["farragut", "karns", "concord"], "huntsville-al": ["madison", "owens-cross-roads", "jones-valley"], "birmingham-al": ["homewood", "vestavia-hills"], "cincinnati-oh": ["hyde-park", "madeira", "indian-hill", "loveland", "blue-ash", "south-dayton"], "northern-ky": ["fort-wright"], }, "lawn-squad": { "chattanooga-tn": ["northshore", "hixson", "ooltewah", "east-brainerd", "signal-mountain"], "knoxville-tn": ["farragut", "karns", "concord"], "huntsville-al": ["madison", "owens-cross-roads", "jones-valley"], "birmingham-al": ["homewood", "vestavia-hills"], "cincinnati-oh": ["hyde-park", "madeira", "indian-hill", "loveland", "blue-ash", "south-dayton"], "northern-ky": ["fort-wright"], }, "new-age-lights": { "chattanooga-tn": ["northshore", "hixson", "ooltewah", "east-brainerd", "signal-mountain"], "knoxville-tn": ["farragut", "karns", "concord"], "huntsville-al": ["madison", "owens-cross-roads", "jones-valley"], "birmingham-al": ["homewood", "vestavia-hills"], "cincinnati-oh": ["hyde-park", "madeira", "indian-hill", "loveland", "blue-ash", "south-dayton"], "northern-ky": ["fort-wright"], }, } SERVICE_KEYWORDS = { "mosquito-squad": ["mosquito-control", "tick-control", "barrier-treatment", "backyard-treatment", "perimeter-spray"], "lawn-squad": ["lawn-fertilization", "aeration", "overseeding", "weed-control", "lawn-treatment"], "new-age-lights": ["permanent-led-lighting", "jellyfish-lighting", "holiday-lighting", "roofline-lighting", "bosso-lighting"], } VALID_EXTENSIONS = {".jpg", ".jpeg", ".png", ".webp", ".heic"} INBOX = LIBRARY_ROOT / "_inbox" UNUSED = LIBRARY_ROOT / "_unused" USED = LIBRARY_ROOT / "_used" DB_PATH = LIBRARY_ROOT / ".photo-tracker.db" # ============================================================ # DATABASE (with auto-migration) # ============================================================ def get_db(): """Connect to SQLite database, create schema, run migrations.""" conn = sqlite3.connect(DB_PATH) conn.row_factory = sqlite3.Row conn.execute(""" CREATE TABLE IF NOT EXISTS photos ( id INTEGER PRIMARY KEY AUTOINCREMENT, filename TEXT NOT NULL, original_filename TEXT, brand TEXT NOT NULL, city TEXT NOT NULL, neighborhood TEXT NOT NULL, service_keyword TEXT, status TEXT NOT NULL DEFAULT 'unused', current_path TEXT NOT NULL, imported_at TEXT NOT NULL, used_at TEXT, used_for TEXT, notes TEXT ) """) # Migration: add file_hash column if missing existing_cols = {r[1] for r in conn.execute("PRAGMA table_info(photos)").fetchall()} if "file_hash" not in existing_cols: conn.execute("ALTER TABLE photos ADD COLUMN file_hash TEXT") conn.execute("CREATE INDEX IF NOT EXISTS idx_status ON photos(status)") conn.execute("CREATE INDEX IF NOT EXISTS idx_brand_city ON photos(brand, city, neighborhood)") conn.execute("CREATE INDEX IF NOT EXISTS idx_hash ON photos(file_hash)") conn.commit() return conn # ============================================================ # HELPERS # ============================================================ def hash_file(path, chunk_size=65536): """Compute SHA-256 hash of a file's contents. Detects true content duplicates.""" h = hashlib.sha256() with open(path, "rb") as f: while True: chunk = f.read(chunk_size) if not chunk: break h.update(chunk) return h.hexdigest() def pick_from_list(prompt, options, allow_skip=False): if not options: print(f"No options available for: {prompt}") return None print(f"\n{prompt}") for i, opt in enumerate(options, 1): print(f" {i}. {opt}") if allow_skip: print(f" 0. (skip)") while True: choice = input("> ").strip() if allow_skip and choice == "0": return None if choice.isdigit() and 1 <= int(choice) <= len(options): return options[int(choice) - 1] print("Invalid choice, try again.") def slugify(text): text = text.lower().strip() text = re.sub(r"[^\w\s-]", "", text) text = re.sub(r"[\s_]+", "-", text) return text.strip("-") def build_filename(brand, city, neighborhood, service_keyword, ext, suffix=None): parts = [service_keyword, neighborhood, city] if suffix: parts.append(suffix) base = "-".join(p for p in parts if p) return f"{base}{ext.lower()}" def ensure_unique_filename(target_dir, filename): target = target_dir / filename if not target.exists(): return filename stem = target.stem ext = target.suffix counter = 2 while (target_dir / f"{stem}-{counter}{ext}").exists(): counter += 1 return f"{stem}-{counter}{ext}" def pretty_date(iso_str): """Turn an ISO timestamp into 'Apr 16, 2026' style.""" if not iso_str: return "never" try: return datetime.fromisoformat(iso_str).strftime("%b %d, %Y") except (ValueError, TypeError): return iso_str[:10] def days_since(iso_str): """How many days since an ISO timestamp. Returns None if not parseable.""" if not iso_str: return None try: dt = datetime.fromisoformat(iso_str) return (datetime.now() - dt).days except (ValueError, TypeError): return None # ============================================================ # SETUP # ============================================================ def cmd_setup(): print(f"Setting up photo library at: {LIBRARY_ROOT}") LIBRARY_ROOT.mkdir(parents=True, exist_ok=True) INBOX.mkdir(exist_ok=True) UNUSED.mkdir(exist_ok=True) USED.mkdir(exist_ok=True) for brand, cities in BRANDS.items(): for city, neighborhoods in cities.items(): for nbhd in neighborhoods: (UNUSED / brand / city / nbhd).mkdir(parents=True, exist_ok=True) (USED / brand / city / nbhd).mkdir(parents=True, exist_ok=True) get_db().close() print("Done. Folder structure created:") print(f" {INBOX} <- drop photos here") print(f" {UNUSED} <- organized unused photos") print(f" {USED} <- auto-moved when marked used") print(f" {DB_PATH} <- tracking database") # ============================================================ # IMPORT (now with duplicate detection) # ============================================================ def cmd_import(): if not INBOX.exists(): print(f"Inbox not found. Run 'setup' first.") return photos = [p for p in INBOX.iterdir() if p.is_file() and p.suffix.lower() in VALID_EXTENSIONS] if not photos: print(f"No photos found in {INBOX}") return print(f"\nFound {len(photos)} photo(s) to import.\n") conn = get_db() imported = 0 skipped_dupes = 0 for idx, photo in enumerate(photos, 1): print(f"\n{'='*60}") print(f"Photo {idx}/{len(photos)}: {photo.name}") print(f"{'='*60}") # Hash first, check for duplicates before we bother the user with questions print("Checking for duplicates...") try: fhash = hash_file(photo) except OSError as e: print(f"Error reading file: {e}. Skipping.") continue existing = conn.execute( "SELECT id, filename, brand, city, neighborhood, status, imported_at " "FROM photos WHERE file_hash = ?", (fhash,) ).fetchone() if existing: print(f"\n⚠ DUPLICATE DETECTED") print(f" This exact file is already in your library:") print(f" Photo #{existing['id']} {existing['filename']}") print(f" Tagged: {existing['brand']} / {existing['city']} / {existing['neighborhood']}") print(f" Status: {existing['status']}, imported {pretty_date(existing['imported_at'])}") print(f"\n What do you want to do?") print(f" 1. Skip and delete the inbox copy (recommended)") print(f" 2. Skip but leave in _inbox/ for now") print(f" 3. Import anyway as a separate entry") choice = input("> ").strip() if choice == "1": photo.unlink() print(" Deleted from inbox.") skipped_dupes += 1 continue elif choice == "2": print(" Left in inbox.") skipped_dupes += 1 continue elif choice != "3": print(" Invalid choice, skipping.") skipped_dupes += 1 continue # else fall through and import anyway # Tag it brand = pick_from_list("Brand:", list(BRANDS.keys())) if not brand: continue city = pick_from_list("City:", list(BRANDS[brand].keys())) if not city: continue neighborhood = pick_from_list("Neighborhood:", BRANDS[brand][city]) if not neighborhood: continue services = SERVICE_KEYWORDS.get(brand, []) service = pick_from_list("Service keyword (or pick custom):", services + ["[custom]"]) if service == "[custom]": service = slugify(input("Enter custom service keyword: ")) suffix = input("Optional suffix (e.g. 'before', 'after', 'twilight') or press enter: ").strip() suffix = slugify(suffix) if suffix else None new_filename = build_filename(brand, city, neighborhood, service, photo.suffix, suffix) target_dir = UNUSED / brand / city / neighborhood target_dir.mkdir(parents=True, exist_ok=True) new_filename = ensure_unique_filename(target_dir, new_filename) target_path = target_dir / new_filename shutil.move(str(photo), str(target_path)) conn.execute(""" INSERT INTO photos (filename, original_filename, brand, city, neighborhood, service_keyword, status, current_path, imported_at, file_hash) VALUES (?, ?, ?, ?, ?, ?, 'unused', ?, ?, ?) """, (new_filename, photo.name, brand, city, neighborhood, service, str(target_path), datetime.now().isoformat(), fhash)) conn.commit() imported += 1 print(f"\n✓ Moved to: {target_path.relative_to(LIBRARY_ROOT)}") conn.close() print(f"\n{'='*60}") print(f"Import complete: {imported} imported, {skipped_dupes} skipped as duplicate.") # ============================================================ # LIST # ============================================================ def cmd_list(brand=None, city=None, neighborhood=None, status="unused"): conn = get_db() query = "SELECT * FROM photos WHERE status = ?" params = [status] if brand: query += " AND brand = ?" params.append(brand) if city: query += " AND city = ?" params.append(city) if neighborhood: query += " AND neighborhood = ?" params.append(neighborhood) query += " ORDER BY brand, city, neighborhood, imported_at" rows = conn.execute(query, params).fetchall() conn.close() if not rows: print(f"No {status} photos match those filters.") return print(f"\n{status.upper()} PHOTOS ({len(rows)} total)") print("=" * 80) current_group = None for r in rows: group = f"{r['brand']} / {r['city']} / {r['neighborhood']}" if group != current_group: print(f"\n[{group}]") current_group = group used_info = f" -> used {r['used_at'][:10]} for '{r['used_for'] or 'GBP'}'" if r['status'] == 'used' else "" imported = f"imported {pretty_date(r['imported_at'])}" print(f" #{r['id']:>4} {r['filename']} ({imported}){used_info}") # ============================================================ # MARK USED # ============================================================ def cmd_mark_used(photo_id, used_for=None): conn = get_db() row = conn.execute("SELECT * FROM photos WHERE id = ?", (photo_id,)).fetchone() if not row: print(f"Photo #{photo_id} not found.") conn.close() return if row['status'] == 'used': print(f"Photo #{photo_id} already marked as used on {row['used_at']}.") conn.close() return if not used_for: used_for = input("What did you use this for? (e.g. 'GBP post 2026-04-16 mosquito hixson'): ").strip() if not used_for: used_for = f"GBP {datetime.now().strftime('%Y-%m-%d')}" src = Path(row['current_path']) dst_dir = USED / row['brand'] / row['city'] / row['neighborhood'] dst_dir.mkdir(parents=True, exist_ok=True) dst = dst_dir / row['filename'] if dst.exists(): dst = dst_dir / ensure_unique_filename(dst_dir, row['filename']) if src.exists(): shutil.move(str(src), str(dst)) else: print(f"Warning: file not found at {src}, updating database only.") conn.execute(""" UPDATE photos SET status = 'used', used_at = ?, used_for = ?, current_path = ? WHERE id = ? """, (datetime.now().isoformat(), used_for, str(dst), photo_id)) conn.commit() conn.close() print(f"✓ #{photo_id} marked used. Moved to: {dst.relative_to(LIBRARY_ROOT)}") # ============================================================ # STATS (now with last upload dates) # ============================================================ def cmd_stats(): conn = get_db() rows = conn.execute(""" SELECT brand, city, neighborhood, status, COUNT(*) as count, MAX(imported_at) as last_import FROM photos GROUP BY brand, city, neighborhood, status ORDER BY brand, city, neighborhood, status """).fetchall() conn.close() if not rows: print("No photos in database yet. Import some first.") return print("\nINVENTORY BY LOCATION") print("=" * 100) print(f"{'Brand':<18} {'City':<18} {'Neighborhood':<20} {'Unused':>7} {'Used':>6} {'Last upload':<14} {'Flags'}") print("-" * 100) grid = {} for r in rows: key = (r['brand'], r['city'], r['neighborhood']) if key not in grid: grid[key] = {'unused': 0, 'used': 0, 'last_import': None} grid[key][r['status']] = r['count'] # Track most recent import across both statuses if r['last_import']: current = grid[key]['last_import'] if not current or r['last_import'] > current: grid[key]['last_import'] = r['last_import'] totals = {'unused': 0, 'used': 0} for (brand, city, nbhd), counts in sorted(grid.items()): unused = counts.get('unused', 0) used = counts.get('used', 0) totals['unused'] += unused totals['used'] += used last = counts['last_import'] last_str = pretty_date(last) days = days_since(last) flags = [] if unused < 3: flags.append("low-stock") if days is not None and days > 60: flags.append(f"stale-{days}d") flag_str = ", ".join(flags) print(f"{brand:<18} {city:<18} {nbhd:<20} {unused:>7} {used:>6} {last_str:<14} {flag_str}") print("-" * 100) print(f"{'TOTAL':<58} {totals['unused']:>7} {totals['used']:>6}") print("\nFlags: 'low-stock' = fewer than 3 unused photos.") print(" 'stale-Nd' = no new photos uploaded in N days (N > 60).") # ============================================================ # LAST UPLOADS (sorted by staleness so worst offenders float up) # ============================================================ def cmd_last_uploads(): conn = get_db() rows = conn.execute(""" SELECT brand, city, neighborhood, MAX(imported_at) as last_import, COUNT(*) as total FROM photos GROUP BY brand, city, neighborhood ORDER BY last_import ASC """).fetchall() conn.close() if not rows: print("No photos in database yet.") return print("\nLAST UPLOAD BY LOCATION (oldest first)") print("=" * 90) print(f"{'Days ago':>9} {'Last upload':<14} {'Brand':<18} {'City':<18} {'Neighborhood':<20} {'Total':>6}") print("-" * 90) for r in rows: days = days_since(r['last_import']) days_str = f"{days}" if days is not None else "?" print(f"{days_str:>9} {pretty_date(r['last_import']):<14} {r['brand']:<18} {r['city']:<18} {r['neighborhood']:<20} {r['total']:>6}") print("\nOldest at top = pockets that need fresh tech photos next.") # ============================================================ # REASSIGN (fix a mislabeled photo) # ============================================================ def cmd_reassign(photo_id): conn = get_db() row = conn.execute("SELECT * FROM photos WHERE id = ?", (photo_id,)).fetchone() if not row: print(f"Photo #{photo_id} not found.") conn.close() return print(f"\nReassigning photo #{photo_id}") print(f" Current filename: {row['filename']}") print(f" Currently tagged: {row['brand']} / {row['city']} / {row['neighborhood']}") print(f" Service keyword: {row['service_keyword']}") print(f" Status: {row['status']}") print("\nPick new tags (or Ctrl+C to cancel):") new_brand = pick_from_list("New brand:", list(BRANDS.keys())) if not new_brand: conn.close() return new_city = pick_from_list("New city:", list(BRANDS[new_brand].keys())) if not new_city: conn.close() return new_nbhd = pick_from_list("New neighborhood:", BRANDS[new_brand][new_city]) if not new_nbhd: conn.close() return # Give option to update service keyword too, since it often differs by brand print(f"\nKeep current service keyword '{row['service_keyword']}' or pick a new one?") services = SERVICE_KEYWORDS.get(new_brand, []) service = pick_from_list("Service keyword:", [f"[keep '{row['service_keyword']}']"] + services + ["[custom]"]) if service and service.startswith("[keep"): new_service = row['service_keyword'] elif service == "[custom]": new_service = slugify(input("Enter custom service keyword: ")) elif service: new_service = service else: new_service = row['service_keyword'] # Build new filename and destination src = Path(row['current_path']) ext = src.suffix if src.suffix else ".jpg" new_filename = build_filename(new_brand, new_city, new_nbhd, new_service, ext) # Destination folder depends on current status (unused vs used) root = USED if row['status'] == 'used' else UNUSED dst_dir = root / new_brand / new_city / new_nbhd dst_dir.mkdir(parents=True, exist_ok=True) new_filename = ensure_unique_filename(dst_dir, new_filename) dst_path = dst_dir / new_filename # Confirm print(f"\nAbout to reassign:") print(f" From: {row['brand']}/{row['city']}/{row['neighborhood']}/{row['filename']}") print(f" To: {new_brand}/{new_city}/{new_nbhd}/{new_filename}") confirm = input("Proceed? (y/n): ").strip().lower() if confirm != "y": print("Cancelled.") conn.close() return # Move the file if src.exists(): shutil.move(str(src), str(dst_path)) else: print(f"Warning: source file not found at {src}. Updating database only.") # Update DB conn.execute(""" UPDATE photos SET brand = ?, city = ?, neighborhood = ?, service_keyword = ?, filename = ?, current_path = ? WHERE id = ? """, (new_brand, new_city, new_nbhd, new_service, new_filename, str(dst_path), photo_id)) conn.commit() conn.close() print(f"✓ Reassigned. New location: {dst_path.relative_to(LIBRARY_ROOT)}") # ============================================================ # DUPLICATE SCAN # ============================================================ def cmd_dupes(): """Scan library for duplicate photos by content hash.""" conn = get_db() # First backfill any missing hashes rows_without_hash = conn.execute( "SELECT id, current_path FROM photos WHERE file_hash IS NULL OR file_hash = ''" ).fetchall() if rows_without_hash: print(f"Backfilling hashes for {len(rows_without_hash)} older photos (one-time)...") for r in rows_without_hash: p = Path(r['current_path']) if p.exists(): try: h = hash_file(p) conn.execute("UPDATE photos SET file_hash = ? WHERE id = ?", (h, r['id'])) except OSError as e: print(f" Could not hash #{r['id']}: {e}") else: print(f" File missing for #{r['id']}: {p}") conn.commit() print("Backfill done.\n") # Find duplicates dupe_groups = conn.execute(""" SELECT file_hash, COUNT(*) as cnt FROM photos WHERE file_hash IS NOT NULL AND file_hash != '' GROUP BY file_hash HAVING cnt > 1 ORDER BY cnt DESC """).fetchall() if not dupe_groups: print("No duplicate photos found. Library is clean.") conn.close() return print(f"\nFound {len(dupe_groups)} set(s) of duplicate photos:") print("=" * 80) for gi, grp in enumerate(dupe_groups, 1): members = conn.execute( "SELECT id, filename, brand, city, neighborhood, status, imported_at " "FROM photos WHERE file_hash = ? ORDER BY imported_at", (grp['file_hash'],) ).fetchall() print(f"\nDuplicate set {gi} ({grp['cnt']} copies, hash {grp['file_hash'][:10]}...):") for m in members: print(f" #{m['id']:>4} [{m['status']}] {m['brand']}/{m['city']}/{m['neighborhood']}/{m['filename']}") print(f" imported {pretty_date(m['imported_at'])}") print("\n" + "=" * 80) print("To remove a duplicate: use 'delete' command (or delete file + row manually).") print("Suggested strategy: keep the oldest 'used' copy (proof of history), delete newer 'unused' copies.") # Offer to interactively clean up print("\nWant to interactively clean up? (y/n)") if input("> ").strip().lower() == "y": for gi, grp in enumerate(dupe_groups, 1): members = conn.execute( "SELECT id, filename, brand, city, neighborhood, status, imported_at, current_path " "FROM photos WHERE file_hash = ? ORDER BY imported_at", (grp['file_hash'],) ).fetchall() print(f"\nSet {gi}:") for i, m in enumerate(members, 1): print(f" {i}. #{m['id']} [{m['status']}] {m['filename']} ({m['brand']}/{m['city']}/{m['neighborhood']})") print(f" 0. Skip this set (leave all copies)") print(f" Enter number(s) to DELETE (comma separated, e.g. '2,3'):") picks = input("> ").strip() if picks == "0" or not picks: continue try: to_delete = [int(x.strip()) for x in picks.split(",")] for pick in to_delete: if 1 <= pick <= len(members): m = members[pick - 1] p = Path(m['current_path']) if p.exists(): p.unlink() conn.execute("DELETE FROM photos WHERE id = ?", (m['id'],)) print(f" Deleted #{m['id']}") conn.commit() except ValueError: print(" Invalid input, skipping this set.") conn.close() # ============================================================ # SEARCH # ============================================================ def cmd_search(term): conn = get_db() rows = conn.execute(""" SELECT * FROM photos WHERE filename LIKE ? OR brand LIKE ? OR city LIKE ? OR neighborhood LIKE ? OR service_keyword LIKE ? OR used_for LIKE ? ORDER BY status, brand, city, neighborhood """, [f"%{term}%"] * 6).fetchall() conn.close() if not rows: print(f"No photos match '{term}'.") return print(f"\nSEARCH RESULTS for '{term}' ({len(rows)} matches)") print("=" * 80) for r in rows: status_tag = "[USED] " if r['status'] == 'used' else "[UNUSED]" print(f"{status_tag} #{r['id']:>4} {r['filename']}") print(f" {r['brand']} / {r['city']} / {r['neighborhood']} (imported {pretty_date(r['imported_at'])})") # ============================================================ # INTERACTIVE MENU # ============================================================ def cmd_menu(): while True: print("\n" + "=" * 50) print("PHOTO SORTER") print("=" * 50) print("1. Import new photos from _inbox/") print("2. List UNUSED photos (filterable)") print("3. Mark photo as used") print("4. Inventory stats (with last upload dates)") print("5. Search photos") print("6. List USED photos (filterable)") print("7. Reassign a photo (fix brand/city/neighborhood)") print("8. Scan library for duplicate photos") print("9. Last upload dates by location") print("0. Quit") choice = input("> ").strip() if choice == "1": cmd_import() elif choice == "2": _filter_and_list("unused") elif choice == "3": pid = input("Photo ID to mark used: ").strip() if pid.isdigit(): cmd_mark_used(int(pid)) elif choice == "4": cmd_stats() elif choice == "5": term = input("Search term: ").strip() if term: cmd_search(term) elif choice == "6": _filter_and_list("used") elif choice == "7": pid = input("Photo ID to reassign: ").strip() if pid.isdigit(): cmd_reassign(int(pid)) elif choice == "8": cmd_dupes() elif choice == "9": cmd_last_uploads() elif choice == "0": print("Done.") break else: print("Invalid choice.") def _filter_and_list(status): print("\nFilter (or press enter to skip):") brand = pick_from_list("Brand (or skip):", list(BRANDS.keys()), allow_skip=True) city = None nbhd = None if brand: city = pick_from_list("City (or skip):", list(BRANDS[brand].keys()), allow_skip=True) if city: nbhd = pick_from_list("Neighborhood (or skip):", BRANDS[brand][city], allow_skip=True) cmd_list(brand=brand, city=city, neighborhood=nbhd, status=status) # ============================================================ # CLI ENTRY # ============================================================ def main(): if len(sys.argv) < 2: print(__doc__) return cmd = sys.argv[1] args = sys.argv[2:] if cmd == "setup": cmd_setup() elif cmd == "import": cmd_import() elif cmd == "list": cmd_list() elif cmd == "used": if not args or not args[0].isdigit(): print("Usage: photo_sorter.py used ") return cmd_mark_used(int(args[0])) elif cmd == "stats": cmd_stats() elif cmd == "search": if not args: print("Usage: photo_sorter.py search ") return cmd_search(" ".join(args)) elif cmd == "reassign": if not args or not args[0].isdigit(): print("Usage: photo_sorter.py reassign ") return cmd_reassign(int(args[0])) elif cmd == "dupes": cmd_dupes() elif cmd == "last-uploads": cmd_last_uploads() elif cmd == "menu": cmd_menu() else: print(f"Unknown command: {cmd}") print(__doc__) if __name__ == "__main__": main()
